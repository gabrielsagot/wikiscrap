import requests
from bs4 import BeautifulSoup
import pandas as pd
import matplotlib.pyplot as plt
import folium
import re

# =====================================================================
# SCRAPING
# =====================================================================
url = "https://fr.wikipedia.org/wiki/Liste_du_patrimoine_mondial_en_France"
headers = {
    'User-Agent': 'Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36'
}

print("=== SCRAPING DES SITES UNESCO FRANCE ===\n")
response = requests.get(url, headers=headers)
response.encoding = 'utf-8'
soup = BeautifulSoup(response.text, 'html.parser')

sites = []
regions = []
types_sites = []
annees = []
coordonnees = []

tableaux = soup.find_all('table', class_='wikitable')
tableau = tableaux[1]  # 2ème tableau = liste des sites

for ligne in tableau.find_all('tr')[1:]:
    cellules = ligne.find_all('td')
    
    if len(cellules) >= 6:
        site = cellules[0].get_text(strip=True)
        region = cellules[1].get_text(strip=True)
        
        # Extraction de l'année
        annee_text = cellules[2].get_text(strip=True)
        annee_match = re.findall(r'\d{4}', annee_text)
        annee = int(annee_match[0]) if annee_match else None
        
        # Extraction du type
        type_text = cellules[4].get_text(strip=True)
        if 'Naturel' in type_text:
            type_s = 'Naturel'
        elif 'Mixte' in type_text:
            type_s = 'Mixte'
        else:
            type_s = 'Culturel'
        
        # ===== AMÉLIORATION : Extraction des coordonnées =====
        # On cherche d'abord dans le lien <a> de la cellule
        coords_link = cellules[5].find('a', class_='external text')
        if coords_link:
            coords_text = coords_link.get_text(strip=True)
        else:
            # Sinon on prend le texte brut
            coords_text = cellules[5].get_text(strip=True)
        
        sites.append(site)
        regions.append(region)
        types_sites.append(type_s)
        annees.append(annee)
        coordonnees.append(coords_text)

df = pd.DataFrame({
    'Site': sites,
    'Region': regions,
    'Type': types_sites,
    'Annee': annees,
    'Coordonnees': coordonnees
})

print(f"✓ {len(df)} sites UNESCO récupérés\n")

# =====================================================================
# AMÉLIORATION : Fonction de conversion DMS vers DD plus robuste
# =====================================================================
def dms2dd(degrees, minutes, seconds, direction):
    """Convertit DMS en degrés décimaux"""
    dd = float(degrees) + float(minutes)/60 + float(seconds)/(60*60)
    if direction in ('S', 'O', 'W', 'sud', 'ouest'):
        dd *= -1
    return dd

def parse_coordinates(coord_string):
    """
    Parse les coordonnées depuis différents formats possibles
    Retourne (latitude, longitude) ou (None, None) si échec
    """
    if not coord_string or coord_string.strip() == '':
        return None, None
    
    try:
        # Format standard : "48° 51′ 29″ N, 2° 17′ 40″ E"
        if ',' in coord_string:
            parts = coord_string.split(',')
            if len(parts) == 2:
                lat_part = parts[0].strip()
                lon_part = parts[1].strip()
                
                # Extraction des chiffres et direction pour latitude
                lat_match = re.findall(r'(\d+)[°\s]+(\d+)[′\'\s]+(\d+)[″"\s]+([NSEO]|nord|sud|est|ouest)', lat_part, re.IGNORECASE)
                if lat_match:
                    lat_deg, lat_min, lat_sec, lat_dir = lat_match[0]
                    lat = dms2dd(lat_deg, lat_min, lat_sec, lat_dir)
                else:
                    # Essayer format décimal direct
                    lat_decimal = re.findall(r'([-]?\d+\.?\d*)', lat_part)
                    if lat_decimal:
                        lat = float(lat_decimal[0])
                    else:
                        return None, None
                
                # Extraction des chiffres et direction pour longitude
                lon_match = re.findall(r'(\d+)[°\s]+(\d+)[′\'\s]+(\d+)[″"\s]+([NSEO]|nord|sud|est|ouest)', lon_part, re.IGNORECASE)
                if lon_match:
                    lon_deg, lon_min, lon_sec, lon_dir = lon_match[0]
                    lon = dms2dd(lon_deg, lon_min, lon_sec, lon_dir)
                else:
                    # Essayer format décimal direct
                    lon_decimal = re.findall(r'([-]?\d+\.?\d*)', lon_part)
                    if lon_decimal:
                        lon = float(lon_decimal[0])
                    else:
                        return None, None
                
                return lat, lon
        
        # Si pas de virgule, peut-être format "lat lon" décimal
        decimals = re.findall(r'([-]?\d+\.?\d+)', coord_string)
        if len(decimals) >= 2:
            return float(decimals[0]), float(decimals[1])
        
    except Exception as e:
        print(f"⚠️  Erreur parsing: {coord_string[:50]}... -> {e}")
        return None, None
    
    return None, None

# =====================================================================

print("Conversion des coordonnées...\n")

latitudes = []
longitudes = []

for i, coord in enumerate(df['Coordonnees']):
    lat, lon = parse_coordinates(coord)
    latitudes.append(lat)
    longitudes.append(lon)

df['Latitude'] = latitudes
df['Longitude'] = longitudes


# === FIX POUR LE SITE SANS COORDONNÉES ===
# Site des mémoriaux WW1 = bien en série, on utilise Notre-Dame-de-Lorette comme point représentatif
df.loc[df['Latitude'].isna(), 'Latitude'] = 50.40
df.loc[df['Longitude'].isna(), 'Longitude'] = 2.71


# =====================================================================
# STATISTIQUES
# =====================================================================
df_map = df.dropna(subset=['Latitude', 'Longitude'])
df_sans_coords = df[df['Latitude'].isna()]

print(f"✓ {len(df_map)} sites AVEC coordonnées")
print(f"✗ {len(df_sans_coords)} sites SANS coordonnées\n")

# =====================================================================

    print(f"{'='*70}")
    print(f"SITES SANS COORDONNÉES (à investiguer)")

    for i, row in df_sans_coords.iterrows():
        print(f"🔴 {row['Site']}")
        print(f"   Région: {row['Region']}")
        print(f"   Coords brutes: '{row['Coordonnees'][:80]}...'")
        print()

print("Génération des graphiques...\n")

# Graphique 1 : Top 10 régions

from folium import plugins

# Carte moderne avec style CartoDB Positron
carte = folium.Map(
    location=[46.6, 2.5], 
    zoom_start=6,
    tiles='CartoDB positron',
    control_scale=True
)

# Configuration des couleurs et icônes par type
type_config = {
    'Culturel': {'color': '#3498db', 'icon': 'fa-landmark'},
    'Naturel': {'color': '#27ae60', 'icon': 'fa-tree'},
    'Mixte': {'color': '#e67e22', 'icon': 'fa-mountain-sun'}
}

# Vérification et ajout des marqueurs
for _, row in df_map.iterrows():
    lat, lon = row['Latitude'], row['Longitude']
    
    # FILTRE : Garder uniquement les points en France métropolitaine et DOM-TOM
    # France métropolitaine : 41°-51° N, -5°-10° E
    # + DOM-TOM : latitude entre -25° et 50°, longitude entre -65° et 60°
    if not ((41 <= lat <= 51 and -5 <= lon <= 10) or  # Métropole
            (-25 <= lat <= -20 and 55 <= lon <= 60) or  # Réunion
            (14 <= lat <= 18 and -63 <= lon <= -60) or  # Guadeloupe/Martinique
            (-23 <= lat <= -21 and 164 <= lon <= 168) or  # Nouvelle-Calédonie
            (-18 <= lat <= -8 and -141 <= lon <= -138) or  # Polynésie française
            (-50 <= lat <= -37 and 50 <= lon <= 78)):  # TAAF
        print(f"⚠️  Coordonnées hors France ignorées : {row['Site']} ({lat}, {lon})")
        continue
    
    config = type_config.get(row['Type'], {'color': '#95a5a6', 'icon': 'fa-map-marker'})
    
    # Popup HTML stylé
    popup_html = f"""
    <div style="font-family: 'Segoe UI', Tahoma, sans-serif; width: 250px;">
        <h4 style="color: {config['color']}; margin: 0 0 10px 0; font-size: 16px; border-bottom: 2px solid {config['color']}; padding-bottom: 5px;">
            <i class="fa {config['icon']}" style="margin-right: 8px;"></i>{row['Site']}
        </h4>
        <p style="margin: 5px 0; color: #555; font-size: 13px;">
            <i class="fa fa-map-pin" style="color: {config['color']}; margin-right: 5px;"></i>
            <strong>Région:</strong> {row['Region']}
        </p>
        <p style="margin: 5px 0; color: #555; font-size: 13px;">
            <i class="fa fa-tag" style="color: {config['color']}; margin-right: 5px;"></i>
            <strong>Type:</strong> {row['Type']}
        </p>
        <p style="margin: 5px 0; color: #555; font-size: 13px;">
            <i class="fa fa-calendar" style="color: {config['color']}; margin-right: 5px;"></i>
            <strong>Inscrit en:</strong> {row['Annee']}
        </p>
    </div>
    """
    
    # Création du marqueur avec icône Font Awesome
    folium.Marker(
        location=[lat, lon],
        popup=folium.Popup(popup_html, max_width=300),
        tooltip=f"<b>{row['Site']}</b>",
        icon=folium.Icon(
            color='white',
            icon_color=config['color'],
            icon=config['icon'],
            prefix='fa'
        )
    ).add_to(carte)
