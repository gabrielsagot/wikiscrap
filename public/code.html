import requests
from bs4 import BeautifulSoup
import pandas as pd
import matplotlib.pyplot as plt
import folium
import re

# =====================================================================
# SCRAPING
# =====================================================================
url = "https://fr.wikipedia.org/wiki/Liste_du_patrimoine_mondial_en_France"
headers = {
    'User-Agent': 'Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36'
}

print("=== SCRAPING DES SITES UNESCO FRANCE ===\n")
response = requests.get(url, headers=headers)
response.encoding = 'utf-8'
soup = BeautifulSoup(response.text, 'html.parser')

sites = []
regions = []
types_sites = []
annees = []
coordonnees = []

tableaux = soup.find_all('table', class_='wikitable')
tableau = tableaux[1]  # 2√®me tableau = liste des sites

for ligne in tableau.find_all('tr')[1:]:
    cellules = ligne.find_all('td')
    
    if len(cellules) >= 6:
        site = cellules[0].get_text(strip=True)
        region = cellules[1].get_text(strip=True)
        
        # Extraction de l'ann√©e
        annee_text = cellules[2].get_text(strip=True)
        annee_match = re.findall(r'\d{4}', annee_text)
        annee = int(annee_match[0]) if annee_match else None
        
        # Extraction du type
        type_text = cellules[4].get_text(strip=True)
        if 'Naturel' in type_text:
            type_s = 'Naturel'
        elif 'Mixte' in type_text:
            type_s = 'Mixte'
        else:
            type_s = 'Culturel'
        
        # ===== AM√âLIORATION : Extraction des coordonn√©es =====
        # On cherche d'abord dans le lien <a> de la cellule
        coords_link = cellules[5].find('a', class_='external text')
        if coords_link:
            coords_text = coords_link.get_text(strip=True)
        else:
            # Sinon on prend le texte brut
            coords_text = cellules[5].get_text(strip=True)
        
        sites.append(site)
        regions.append(region)
        types_sites.append(type_s)
        annees.append(annee)
        coordonnees.append(coords_text)

df = pd.DataFrame({
    'Site': sites,
    'Region': regions,
    'Type': types_sites,
    'Annee': annees,
    'Coordonnees': coordonnees
})

print(f"‚úì {len(df)} sites UNESCO r√©cup√©r√©s\n")

# AM√âLIORATION : Fonction de conversion DMS vers DD plus robuste
# =====================================================================
def dms2dd(degrees, minutes, seconds, direction):
    """Convertit DMS en degr√©s d√©cimaux"""
    dd = float(degrees) + float(minutes)/60 + float(seconds)/(60*60)
    if direction in ('S', 'O', 'W', 'sud', 'ouest'):
        dd *= -1
    return dd

def parse_coordinates(coord_string):
    """
    Parse les coordonn√©es depuis diff√©rents formats possibles
    Retourne (latitude, longitude) ou (None, None) si √©chec
    """
    if not coord_string or coord_string.strip() == '':
        return None, None
    
    try:
        # Format standard : "48¬∞ 51‚Ä≤ 29‚Ä≥ N, 2¬∞ 17‚Ä≤ 40‚Ä≥ E"
        if ',' in coord_string:
            parts = coord_string.split(',')
            if len(parts) == 2:
                lat_part = parts[0].strip()
                lon_part = parts[1].strip()
                
                # Extraction des chiffres et direction pour latitude
                lat_match = re.findall(r'(\d+)[¬∞\s]+(\d+)[‚Ä≤\'\s]+(\d+)[‚Ä≥"\s]+([NSEO]|nord|sud|est|ouest)', lat_part, re.IGNORECASE)
                if lat_match:
                    lat_deg, lat_min, lat_sec, lat_dir = lat_match[0]
                    lat = dms2dd(lat_deg, lat_min, lat_sec, lat_dir)
                else:
                    # Essayer format d√©cimal direct
                    lat_decimal = re.findall(r'([-]?\d+\.?\d*)', lat_part)
                    if lat_decimal:
                        lat = float(lat_decimal[0])
                    else:
                        return None, None
                
                # Extraction des chiffres et direction pour longitude
                lon_match = re.findall(r'(\d+)[¬∞\s]+(\d+)[‚Ä≤\'\s]+(\d+)[‚Ä≥"\s]+([NSEO]|nord|sud|est|ouest)', lon_part, re.IGNORECASE)
                if lon_match:
                    lon_deg, lon_min, lon_sec, lon_dir = lon_match[0]
                    lon = dms2dd(lon_deg, lon_min, lon_sec, lon_dir)
                else:
                    # Essayer format d√©cimal direct
                    lon_decimal = re.findall(r'([-]?\d+\.?\d*)', lon_part)
                    if lon_decimal:
                        lon = float(lon_decimal[0])
                    else:
                        return None, None
                
                return lat, lon
        
        # Si pas de virgule, peut-√™tre format "lat lon" d√©cimal
        decimals = re.findall(r'([-]?\d+\.?\d+)', coord_string)
        if len(decimals) >= 2:
            return float(decimals[0]), float(decimals[1])
        
    except Exception as e:
        print(f"‚ö†Ô∏è  Erreur parsing: {coord_string[:50]}... -> {e}")
        return None, None
    
    return None, None
